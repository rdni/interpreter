print("Calculator");

var num1 = int(input("Enter number 1: "));
var num2 = int(input("Enter number 2: "));
var operator = input("Enter operator: ");

var answer;
if operator == "+" {
    answer = num1 + num2;
} else if operator == "-" {
    answer = num1 - num2;
} else if operator == "*" {
    answer = num1 * num2;
} else if operator == "/" {
    answer = num1 / num2;
} else {
    print("Invalid operator");
    exit();
}

print("The answer is " + answer + ".");
